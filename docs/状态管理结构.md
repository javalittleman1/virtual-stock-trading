| 文档版本 | 修改日期   | 修改内容                 |
| -------- | ---------- | ------------------------ |
| V1.0     | 2026-04-09 | 初稿，定义 MVP 核心 Store |

---

## 1. 概述

- **状态管理库**：Zustand（轻量、无模板代码、支持 TypeScript）。
- **持久化**：用户偏好（如主题）使用 `persist` 中间件存入 `localStorage`；敏感数据（如持仓、资金）不持久化，每次会话从服务端获取。
- **实时更新**：通过 Supabase Realtime 订阅数据库变更，在回调中调用 Store 的 `set` 方法更新状态。
- **Store 划分原则**：按功能域拆分，避免单一巨型 Store。

## 2. Store 目录结构

```
stores/
├── useAuthStore.ts        # 用户认证状态
```

## 3. 详细 Store 定义

### 3.1 useAuthStore（用户认证状态）

> **MVP 范围**：P0

**职责**：管理 Supabase 会话状态，提供登录/退出方法。

```typescript
import { create } from 'zustand';
import { supabase } from '@/lib/supabase';

interface AuthState {
  session: Session | null;
  user: User | null;
  isLoading: boolean;
  setSession: (session: Session | null) => void;
  signIn: (email: string, password: string) => Promise<void>;
  signOut: () => Promise<void>;
  initialize: () => void;
}

export const useAuthStore = create<AuthState>((set) => ({
  session: null,
  user: null,
  isLoading: true,
  setSession: (session) => set({ session, user: session?.user ?? null }),
  signIn: async (email, password) => {
    const { data, error } = await supabase.auth.signInWithPassword({ email, password });
    if (error) throw error;
    set({ session: data.session, user: data.user });
  },
  signOut: async () => {
    await supabase.auth.signOut();
    set({ session: null, user: null });
  },
  initialize: () => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      set({ session, user: session?.user ?? null, isLoading: false });
    });
    supabase.auth.onAuthStateChange((_event, session) => {
      set({ session, user: session?.user ?? null });
    });
  },
}));
```

### 3.2 useUserStore（用户资料与资金）

> **MVP 范围**：P0

**职责**：管理用户虚拟资金、账户概览数据，支持资金更新（交易后）。

```typescript
import { create } from 'zustand';
import { supabase } from '@/lib/supabase';

interface Profile {
  id: string;
  email: string;
  virtual_balance: number;
}

interface UserState {
  profile: Profile | null;
  isLoading: boolean;
  fetchProfile: (userId: string) => Promise<void>;
  updateBalance: (newBalance: number) => void;
  subscribeBalance: (userId: string) => () => void;
}

export const useUserStore = create<UserState>((set, get) => ({
  profile: null,
  isLoading: false,
  fetchProfile: async (userId) => {
    set({ isLoading: true });
    const { data } = await supabase
      .from('profiles')
      .select('*')
      .eq('id', userId)
      .single();
    set({ profile: data, isLoading: false });
  },
  updateBalance: (newBalance) => {
    const { profile } = get();
    if (profile) {
      set({ profile: { ...profile, virtual_balance: newBalance } });
    }
  },
  subscribeBalance: (userId) => {
    const channel = supabase
      .channel(`profile-${userId}`)
      .on(
        'postgres_changes',
        { event: 'UPDATE', schema: 'public', table: 'profiles', filter: `id=eq.${userId}` },
        (payload) => {
          set({ profile: payload.new as Profile });
        }
      )
      .subscribe();
    return () => channel.unsubscribe();
  },
}));
```

### 3.3 useStockStore（行情数据与自选股）

> **MVP 范围**：P0

**职责**：管理股票列表、实时行情、自选股列表，支持搜索和 Realtime 订阅。

```typescript
import { create } from 'zustand';

interface Stock {
  symbol: string;
  name: string;
  current_price: number;
  change_percent: number;
  updated_at: string;
}

interface StockState {
  stocks: Stock[];
  watchlist: Stock[];
  searchKeyword: string;
  setSearchKeyword: (kw: string) => void;
  fetchStocks: (keyword?: string) => Promise<void>;
  fetchWatchlist: (userId: string) => Promise<void>;
  addToWatchlist: (userId: string, symbol: string) => Promise<void>;
  removeFromWatchlist: (userId: string, symbol: string) => Promise<void>;
  subscribePrices: (symbols?: string[]) => () => void;
  updateStockPrice: (symbol: string, price: number) => void;
}

export const useStockStore = create<StockState>((set, get) => ({
  stocks: [],
  watchlist: [],
  searchKeyword: '',
  setSearchKeyword: (kw) => set({ searchKeyword: kw }),
  fetchStocks: async (keyword) => {
    const params = new URLSearchParams();
    if (keyword) params.append('q', keyword);
    const res = await fetch(`/api/stocks?${params}`);
    const { data } = await res.json();
    set({ stocks: data });
  },
  fetchWatchlist: async (userId) => {
    const res = await fetch('/api/watchlist');
    const { data } = await res.json();
    set({ watchlist: data });
  },
  addToWatchlist: async (userId, symbol) => {
    await fetch('/api/watchlist', {
      method: 'POST',
      body: JSON.stringify({ symbol }),
    });
    get().fetchWatchlist(userId);
  },
  removeFromWatchlist: async (userId, symbol) => {
    await fetch(`/api/watchlist/${symbol}`, { method: 'DELETE' });
    get().fetchWatchlist(userId);
  },
  subscribePrices: (symbols) => {
    const filter = symbols?.length ? `symbol=in.(${symbols.join(',')})` : undefined;
    const channel = supabase
      .channel('stocks-realtime')
      .on('postgres_changes', { event: 'UPDATE', schema: 'public', table: 'stocks', filter }, (payload) => {
        get().updateStockPrice(payload.new.symbol, payload.new.current_price);
      })
      .subscribe();
    return () => channel.unsubscribe();
  },
  updateStockPrice: (symbol, price) => {
    set((state) => ({
      stocks: state.stocks.map(s => s.symbol === symbol ? { ...s, current_price: price } : s),
      watchlist: state.watchlist.map(s => s.symbol === symbol ? { ...s, current_price: price } : s),
    }));
  },
}));
```

### 3.4 useTradeStore（交易委托与持仓）

> **MVP 范围**：P0

**职责**：管理用户持仓、委托订单，提供下单、撤单方法，并同步更新资金。

```typescript
import { create } from 'zustand';

interface Holding {
  symbol: string;
  name: string;
  quantity: number;
  avg_cost: number;
  current_price: number;
  market_value: number;
  profit_loss: number;
  profit_loss_percent: number;
}

interface Order {
  order_id: string;
  symbol: string;
  type: 'buy' | 'sell';
  price: number;
  quantity: number;
  filled_quantity: number;
  status: 'pending' | 'filled' | 'partial' | 'cancelled';
  created_at: string;
}

interface TradeState {
  holdings: Holding[];
  orders: Order[];
  isLoading: boolean;
  fetchHoldings: () => Promise<void>;
  fetchOrders: (status?: string) => Promise<void>;
  submitOrder: (order: { symbol: string; type: string; price: number; quantity: number }) => Promise<void>;
  cancelOrder: (orderId: string) => Promise<void>;
  subscribeHoldings: (userId: string) => () => void;
}

export const useTradeStore = create<TradeState>((set, get) => ({
  holdings: [],
  orders: [],
  isLoading: false,
  fetchHoldings: async () => {
    const res = await fetch('/api/trade/positions');
    const { data } = await res.json();
    set({ holdings: data });
  },
  fetchOrders: async (status) => {
    const url = status ? `/api/trade/orders?status=${status}` : '/api/trade/orders';
    const res = await fetch(url);
    const { data } = await res.json();
    set({ orders: data });
  },
  submitOrder: async (order) => {
    const res = await fetch('/api/trade/order', {
      method: 'POST',
      body: JSON.stringify(order),
    });
    const result = await res.json();
    if (result.status === 'filled') {
      get().fetchHoldings();
      useUserStore.getState().fetchProfile(useAuthStore.getState().user!.id);
    }
    get().fetchOrders();
  },
  cancelOrder: async (orderId) => {
    await fetch(`/api/trade/order/${orderId}`, { method: 'DELETE' });
    get().fetchOrders();
  },
  subscribeHoldings: (userId) => {
    const channel = supabase
      .channel(`portfolios-${userId}`)
      .on('postgres_changes', { event: '*', schema: 'public', table: 'portfolios', filter: `user_id=eq.${userId}` }, () => {
        get().fetchHoldings();
      })
      .subscribe();
    return () => channel.unsubscribe();
  },
}));
```

### 3.5 useAnalysisStore（策略分析模块）

> **MVP 范围**：P1

**职责**：管理首板观察池、五大回调信号、预警配置。

```typescript
import { create } from 'zustand';

interface FirstBoardStock {
  symbol: string;
  name: string;
  concepts: string[];
  seal_ratio: number;
}

interface Signal {
  signal_type: string;
  name: string;
  symbol: string;
  stock_name: string;
  support_price?: number;
  signal_active: boolean;
}

interface AlertConfig {
  symbol: string;
  pressure_break: boolean;
  tail_confirm: boolean;
  stop_loss_percent: number;
}

interface AnalysisState {
  firstBoard: FirstBoardStock[];
  signals: Signal[];
  alertConfigs: Record<string, AlertConfig>;
  fetchFirstBoard: () => Promise<void>;
  fetchSignals: (type?: string) => Promise<void>;
  subscribeAlert: (config: AlertConfig) => Promise<void>;
}

export const useAnalysisStore = create<AnalysisState>((set) => ({
  firstBoard: [],
  signals: [],
  alertConfigs: {},
  fetchFirstBoard: async () => {
    const res = await fetch('/api/analysis/first-board');
    const { data } = await res.json();
    set({ firstBoard: data });
  },
  fetchSignals: async (type) => {
    const url = type ? `/api/analysis/signals?signal_type=${type}` : '/api/analysis/signals';
    const res = await fetch(url);
    const { data } = await res.json();
    set({ signals: data });
  },
  subscribeAlert: async (config) => {
    await fetch('/api/analysis/alert/subscribe', {
      method: 'POST',
      body: JSON.stringify(config),
    });
    set((state) => ({
      alertConfigs: { ...state.alertConfigs, [config.symbol]: config },
    }));
  },
}));
```

### 3.6 useUIStore（UI 状态）

> **MVP 范围**：P0（主题）、P1（其他）

**职责**：管理主题、侧边栏折叠、弹窗显隐、Toast 消息。

```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface UIState {
  theme: 'light' | 'dark';
  sidebarCollapsed: boolean;
  activeModal: string | null;
  toastMessage: string | null;
  setTheme: (theme: 'light' | 'dark') => void;
  toggleSidebar: () => void;
  openModal: (id: string) => void;
  closeModal: () => void;
  showToast: (msg: string) => void;
  hideToast: () => void;
}

export const useUIStore = create<UIState>()(
  persist(
    (set) => ({
      theme: 'light',
      sidebarCollapsed: false,
      activeModal: null,
      toastMessage: null,
      setTheme: (theme) => set({ theme }),
      toggleSidebar: () => set((s) => ({ sidebarCollapsed: !s.sidebarCollapsed })),
      openModal: (id) => set({ activeModal: id }),
      closeModal: () => set({ activeModal: null }),
      showToast: (msg) => set({ toastMessage: msg }),
      hideToast: () => set({ toastMessage: null }),
    }),
    { name: 'ui-storage', partialize: (state) => ({ theme: state.theme }) }
  )
);
```

## 4. Store 间协作模式

| 场景                 | 涉及 Store                        | 协作方式                                                     |
| -------------------- | --------------------------------- | ------------------------------------------------------------ |
| 用户登录后初始化     | authStore → userStore, stockStore | 登录成功后调用各 Store 的 fetch 方法拉取数据，并建立 Realtime 订阅 |
| 下单成功后更新资金   | tradeStore → userStore            | `submitOrder` 完成后调用 `userStore.fetchProfile` 刷新余额    |
| 实时行情推送         | supabase → stockStore             | `subscribePrices` 回调中直接 `set` 更新 stockStore 中的价格   |
| 自选股变更后刷新列表 | stockStore（自更新）              | 调用 API 后重新 `fetchWatchlist`                             |
| 主题切换             | uiStore + DOM 操作                | `setTheme` 同时在 `document.documentElement` 上切换 class     |

---

## 5. 与 Supabase Realtime 集成模式

所有需要实时更新的 Store 均提供 `subscribeXxx` 方法，返回清理函数。在页面或布局组件中统一管理订阅生命周期：

```tsx
// app/(dashboard)/layout.tsx
useEffect(() => {
  const userId = auth.user?.id;
  if (!userId) return;
  
  const unsubBalance = useUserStore.getState().subscribeBalance(userId);
  const unsubHoldings = useTradeStore.getState().subscribeHoldings(userId);
  const unsubPrices = useStockStore.getState().subscribePrices();
  
  return () => {
    unsubBalance();
    unsubHoldings();
    unsubPrices();
  };
}, [auth.user]);
```
